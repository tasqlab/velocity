// ── TABS ──
function setTab(el) {
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
}

// ── FILTER CONVS ──
function filterConvs(val) {
  document.querySelectorAll('.conv-item').forEach(c => {
    const name = c.querySelector('.conv-name')?.textContent.toLowerCase() || '';
    c.style.display = val && !name.includes(val.toLowerCase()) ? 'none' : '';
  });
}

// ── OPEN CONV ──
function openConv(el, name, avatar, bg, status, preview, time, sub) {
  document.querySelectorAll('.conv-item').forEach(c => c.classList.remove('active'));
  el.classList.add('active');

  document.getElementById('chatName').textContent = name;
  document.getElementById('chatSub').textContent = sub;
  const chatAv = document.getElementById('chatAvatar');
  chatAv.textContent = avatar;
  chatAv.style.background = bg;

  const pulse = document.querySelector('.online-pulse');
  if (status === 'online') { pulse.style.display = 'block'; }
  else { pulse.style.display = 'none'; }

  document.querySelector('.composer-input').placeholder = `Message ${name.split(' ')[0]}`;

  // simulate clearing + showing new chat
  document.getElementById('newMessages').innerHTML = '';
}

// ── REACT TOGGLE ──
function toggleReact(el) {
  el.classList.toggle('on');
  const count = el.querySelector('.react-count');
  count.textContent = parseInt(count.textContent) + (el.classList.contains('on') ? 1 : -1);
}

// ── REPLY ──
document.querySelectorAll('.bubble').forEach(b => {
  b.addEventListener('dblclick', () => {
    document.getElementById('replyPreview').classList.add('show');
    document.getElementById('msgInput').focus();
  });
});
function closeReply() { document.getElementById('replyPreview').classList.remove('show'); }

function autoResize(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 110) + 'px';
}

function getTime() {
  return new Date().toLocaleTimeString('en-US',{hour:'numeric',minute:'2-digit',hour12:true});
}
function esc(t) {
  return t.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function sendMsg() {
  const input = document.getElementById('msgInput');
  const text = input.value.trim();
  if (!text) return;

  const area = document.getElementById('newMessages');

  // Sent bubble
  const row = document.createElement('div');
  row.className = 'brow out new-group';
  row.innerHTML = `
    <div>
      <div class="bubble bubble-new">
        <div class="bubble-text">${esc(text)}</div>
        <div class="bubble-footer">
          <span class="bubble-time">${getTime()}</span>
          <span class="bubble-ticks">✓</span>
        </div>
      </div>
    </div>`;
  area.appendChild(row);

  input.value = ''; input.style.height = 'auto';
  closeReply();
  scrollBottom();

  // Tick upgrade
  setTimeout(() => {
    const ticks = row.querySelector('.bubble-ticks');
    if(ticks) ticks.textContent = '✓✓';
  }, 800);

  // Show typing
  const typingRow = document.getElementById('typingRow');
  typingRow.style.display = 'flex';
  scrollBottom();

  // Reply
  setTimeout(() => {
    typingRow.style.display = 'none';

    const chatName = document.getElementById('chatName').textContent;
    const senderKey = Object.keys(AVATARS).find(k => chatName.includes(k)) || 'nova_design';
    const av = AVATARS[senderKey];
    const reply = REPLIES[Math.floor(Math.random() * REPLIES.length)];

    const replyRow = document.createElement('div');
    replyRow.className = 'brow in new-group';
    replyRow.innerHTML = `
      <div class="in-avatar">${av.e}</div>
      <div>
        <div class="bubble bubble-new" style="background:${av.bg.replace('0.18','0.15')}">
          <div class="bubble-text">${reply}</div>
          <div class="bubble-footer"><span class="bubble-time">${getTime()}</span></div>
        </div>
      </div>`;
    area.appendChild(replyRow);
    scrollBottom();

    // Update conv preview
    document.querySelector('.conv-item.active .conv-preview').textContent = reply;
    document.querySelector('.conv-item.active .conv-time').textContent = getTime();
  }, 1200 + Math.random() * 800);
}

function handleKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMsg(); }
}

function scrollBottom() {
  const a = document.getElementById('msgsArea');
  a.scrollTop = a.scrollHeight;
}

function showNewChatHint() {
  // flash the search box
  const sb = document.querySelector('.search-box');
  sb.style.borderColor = 'rgba(59,130,246,0.5)';
  sb.style.boxShadow = '0 0 0 3px rgba(37,99,235,0.1)';
  document.querySelector('.search-input').focus();
  setTimeout(() => { sb.style.borderColor = ''; sb.style.boxShadow = ''; }, 1200);
}

// scroll to bottom on load
window.addEventListener('load', scrollBottom);
